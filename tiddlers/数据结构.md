### 哈希表数据库

```
struct value {
    int ref_count;
    size_t size;
    char data[];
};

struct entry {
    char name[32];
    struct entry *next;
    struct value *value;
};

struct tx_record {
    char name[32];
    struct value *old_value;
    struct value *new_value;
    int is_update;
};

struct database {
    struct entry *buckets[64];
    struct tx_record *tx;
    int in_tx;
};
```
